Do the same thing to scrap the values for all the other players with a loop¶

In [11]:
#Lists to store the scraped values
names = []
options = []
team2018_2019 = []
agent = []
In [12]:
#Loop to grab all the free agents
for container in player_containers:
    
    #Get the name for the player and put it in the list created above
    name = container.a.text
    names.append(name)

    #Get the option for the player and put it in the list created above
    option = container.b.text
    options.append(option)

    #Get the team for the player and put it in the list created above
    team = container.find_all('b')[5].text[6:]
    team2018_2019.append(team)

    #Get the agent name for the player and put it in the list created above
    agents = container.find('div', attrs={'style':'display:inline-block;width:32%;'}).text[7:]
    agent.append(agents)
In [13]:
#Put it in a dataframe with pandas
import pandas as pd

df = pd.DataFrame({'Player': names,
                  'Option': options,
                  'Team2018_2019': team2018_2019,
                  'Agent': agent})
